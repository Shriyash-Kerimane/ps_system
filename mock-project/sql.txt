create database accountdb;

use accountdb;

create table accounts (accountNumber integer primary key, accountType varchar(15), balance double, customerId integer , status varchar(15), branch varchar(15));

create table statement (transactionId integer primary key,accountNumber integer, transactionType varchar(15), amount double, balance double, date varchar(15));

create table customers (customerId integer primary key, customerName varchar(30), phoneNumber varchar(15), emailId varchar(40));

create database credentialdb;

use credentialdb;

create table users(userid integer primary key, username varchar(30), password varchar(15),user varchar(15));
